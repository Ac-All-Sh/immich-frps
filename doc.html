  acallsh/frps-panel:latest

# жЎҘжҺҘжЁЎејҸbridge mode

  -p 7000:7000 -p 7500:7500 -p 8080:8080 \
  -v /etc/frp:/etc/frp acallsh/frps-panel:latest

docker-compose.yml

version: '3.8'
services:
  frps-panel:
    image: acallsh/frps-panel:latest
    container_name: frps-panel
    network_mode: host
    restart: unless-stopped
    volumes:
      - /etc/frp:/etc/frp
      - /var/log/frp:/var/log/frp
